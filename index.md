    <script>
    (function() {
        // 内置星空效果 —— 闪烁星星 (模仿star.js)
        function initStars() {
            var sky = document.getElementById('skyContainer');
            if (!sky) return;

            // 创建canvas
            var canvas = document.createElement('canvas');
            canvas.classList.add('stars-canvas');
            sky.appendChild(canvas);

            var ctx = canvas.getContext('2d');
            var width, height;
            var stars = [];
            var STAR_COUNT = 150; // 星星数量
            var twinkleSpeed = 0.005; // 闪烁速度

            // 生成随机星星
            function generateStars() {
                stars = [];
                for (var i = 0; i < STAR_COUNT; i++) {
                    stars.push({
                        x: Math.random(),
                        y: Math.random(),
                        radius: Math.random() * 2 + 1, // 半径 1~3px
                        baseAlpha: Math.random() * 0.7 + 0.3, // 基础亮度 0.3~1.0
                        speed: Math.random() * 0.02 + 0.005, // 闪烁速度
                        phase: Math.random() * 2 * Math.PI // 随机相位
                    });
                }
            }

            // 调整canvas尺寸
            function resizeCanvas() {
                width = sky.clientWidth;
                height = sky.clientHeight;
                canvas.width = width;
                canvas.height = height;
            }

            // 绘制星星 (带闪烁效果)
            function drawStars(timestamp) {
                if (!ctx || width === 0 || height === 0) return;

                ctx.clearRect(0, 0, width, height);

                // 时间因子用于闪烁
                var time = (timestamp || 0) * twinkleSpeed;

                stars.forEach(function(s) {
                    // 动态透明度：基于基础亮度和正弦波动
                    var twinkle = Math.sin(time * 10 + s.phase) * 0.3 + 0.7; // 0.4~1.0
                    var alpha = s.baseAlpha * twinkle;
                    alpha = Math.min(1, Math.max(0.2, alpha)); // 限制范围

                    ctx.fillStyle = 'rgba(255, 255, 255, ' + alpha + ')';
                    ctx.beginPath();
                    ctx.arc(s.x * width, s.y * height, s.radius, 0, Math.PI * 2);
                    ctx.fill();
                });

                requestAnimationFrame(drawStars);
            }

            // 监听窗口大小变化
            window.addEventListener('resize', function() {
                resizeCanvas();
            });

            // 初始化
            generateStars();
            resizeCanvas();

            // 开始动画
            requestAnimationFrame(drawStars);
        }

        // 等待DOM完全加载
        if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', initStars);
        } else {
            initStars();
        }
    })();
    </script>
